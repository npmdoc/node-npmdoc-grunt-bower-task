  grunt.initConfig({
    jshint: {
      all: [ 'Gruntfile.js', 'tasks/**/*.js', 'test/**/*_test.js', 'specs/**/*.js' ],
      options: {
        jshintrc: '.jshintrc'
      }
    },

    bower: {
      install: {
        options: {
          cleanTargetDir: true,
          cleanBowerDir: true,
          install: true,
          copy: true
        }
      },
      cleanup: {
        options: {
          cleanTargetDir: true,
          cleanBowerDir: true,
          install: false,
          copy: false
        }
      }
    },

    nodeunit: {
      tests: ['test/*_test.js']
    },

    simplemocha: {
      options: {
        timeout: '2s',
        reporter: 'nyan'
      },

      all: { src: ['specs/**/*.js'] }
    }


  });

  grunt.loadTasks('tasks');

  grunt.loadNpmTasks('grunt-contrib-jshint');
  grunt.loadNpmTasks('grunt-simple-mocha');
  grunt.loadNpmTasks('grunt-contrib-nodeunit');

  grunt.registerTask('test', ['jshint', 'nodeunit', 'simplemocha']);

  grunt.registerTask('default', ['test']);

}

    getLayout = function (layout, fail) {
  if (_.isFunction(layout)) {
    return layout;
  }

  if (!_.isString(layout)) {
    fail('Layout should be specified by name or as a function');
  }

  if (_(defaultLayouts).has(layout)) {
    return defaultLayouts[layout];
  }

  fail('The following named layouts are supported: ' + _.keys(defaultLayouts).join(', '));
}

function grunt-bower-task.package_matcher.matches (pkg, override_key)

  • description and source-code
    matches = function (pkg, override_key) {
  if (pkg === override_key) {
    return true;
  }

  if (override_key.match(/^\/.*\/$/)) {
    var trueRegexMatcher = new RegExp(override_key.replace(/^\/|\/$/g, ''));
    return trueRegexMatcher.test(pkg);
  }

  if (override_key.indexOf('*') >= 0) {
    override_key = override_key.replace(/[\-\[\]\/\{\}\(\)\+\?\.\\\^\$\|]/g, '\\$&').replace('*', '.+');
    var wildcardMatcher = new RegExp(override_key);
    return wildcardMatcher.test(pkg);
  }

  return false;
}
